CREATE TABLE Ticket
(
id int,
owner int,
project int,
created date,
title varchar(40),
state int,
type int,
description text,
PRIMARY KEY (id)
);

(
id int,
name varchar,
hash varchar,
password varchar,
salt varchar,
email varchar,
admin bool,
PRIMARY KEY (id)
);
